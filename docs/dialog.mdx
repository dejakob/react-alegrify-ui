## Example

<Playground>
    {() => {
        class Example extends React.Component {
            componentWillMount() {
                this.setState({ show: false });
            }

            render() {
                return (
                    <React.Fragment>
                        <Button
                            primary
                            onClick={() => this.setState({ show: true })}
                        >
                            Show modal
                        </Button>

                        <Dialog
                            open={this.state.show}
                        >
                            Some dialog content

                            <Button
                                full
                                primary
                                onClick={() => this.setState({ show: false })}
                            >
                                Dismiss
                            </Button>
                        </Dialog>
                    </React.Fragment>
                );
            }
        }
        return <Example />
    }}
    
</Playground>
